Consolidating similar job titles into one, and grad school focuses into one

alum <- rename(alum, job_title = `What is your current job position / title?`,
               grad_trade_pro_degree = `In what area is the degree you obtained / are obtaining ?`)

alum <- alum %>% mutate(
  job = case_when(
    grepl('Data Analyst|Analyst|Data Analytic', job_title) ~ "Data Analyst",
    grepl('Data Scien', job_title)  ~ "Data Scientist",
    grepl('Consult|consult', job_title) ~ "Consultant",
    grepl('Business Analy', job_title) ~ "Business Analyst",
    grepl('Teacher|Profess|Teaching', job_title) ~ "Academic Instructor",
    grepl('Biostat', job_title) ~ "Biostatistician",
    grepl('Engineer', job_title) ~ "Software Engineer",
    grepl('Statistician|statistician', job_title) ~ "Statistician",
    grepl("Actuar|actuar", job_title) ~ "Actuary",
    grepl("Program", job_title) ~ "Programmer",
    grepl("Research", job_title) ~ "Researcher",
    grepl("Software", job_title) ~ "Software Developer",
    TRUE ~ job_title
    
   )
  )

alum <- alum %>% mutate(
  masters = grad_trade_pro_degree,
  masters = case_when(
    grepl("Biostat", grad_trade_pro_degree) ~ "Biostatistics",
    grepl("Statistics", grad_trade_pro_degree) ~ "Statistics",
    grepl("Education", grad_trade_pro_degree) ~ "Education",
    grepl("Business", grad_trade_pro_degree) ~ "Business",
    grepl("Informatics|informatics", grad_trade_pro_degree) ~ "Informatics",
    TRUE ~ grad_trade_pro_degree
  )
)

alum$`Capstone Experiences` <- as.vector(alum$`Capstone Experiences`)

prep_sub <- stack(alum[33:35])

alum$Coursework <- factor(alum$Coursework, levels = c("Very unprepared", "Somewhat unprepared", "Neutral", "Somewhat prepared", "Very prepared"))
alum$`Statistical Software` <- factor(alum$`Statistical Software`, levels = c("Very unprepared", "Somewhat unprepared", "Neutral", "Somewhat prepared", "Very prepared"))

ggplot(data = prep_sub, aes(x = ind, fill = values)) +
  geom_bar(stat = "count", position = position_dodge(), alpha = 0.75, color = "white") +
  xlab("") + ylab("Count") + labs(fill = "Level of Preparedness")
